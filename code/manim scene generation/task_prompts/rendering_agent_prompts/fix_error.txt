class Scene_Helper:
    def __init__(self, scene: Scene):
        self.scene = scene

    def fix_transcript(self, text: str) -> str:
        s = (text or "").replace("\n", " ").strip()
        s = " ".join(s.split())
        if not s:
            return s
        if s[0].isalpha():
            s = s[0].upper() + s[1:]
        if s[-1].isalnum():
            s += "."
        s = re.sub(r"([,.!?])(\w)", r"\1 \2", s)
        return s

    def normalize_one_line(self, text: str) -> str:
        return " ".join((text or "").replace("\n", " ").split())

    def safe_rect(self) -> Rectangle:
        r = Rectangle(width=config.frame_width - 1.0, height=config.frame_height - 1.0)
        r.set_stroke(GREY_B, width=1).set_fill(opacity=0)
        return r

    # --------- ManimCE v0.19.0 SAFE AABB (NO get_bounding_box / bounding_box) ----------
    def _aabb_xy(self, m: Mobject, pad: float = 0.0) -> Tuple[float, float, float, float]:
        try:
            pts = np.array(m.get_all_points())
        except Exception:
            pts = np.empty((0, 3))

        if pts.size == 0:
            c = np.array(m.get_center())
            w = float(getattr(m, "width", 0.0))
            h = float(getattr(m, "height", 0.0))
            x0, x1 = float(c[0] - w / 2), float(c[0] + w / 2)
            y0, y1 = float(c[1] - h / 2), float(c[1] + h / 2)
        else:
            x0, x1 = float(pts[:, 0].min()), float(pts[:, 0].max())
            y0, y1 = float(pts[:, 1].min()), float(pts[:, 1].max())

        return (x0 - pad, x1 + pad, y0 - pad, y1 + pad)

    def inside_safe_area(self, m: Mobject, margin: float = 0.5) -> bool:
        left = -config.frame_width / 2 + margin
        right = config.frame_width / 2 - margin
        bottom = -config.frame_height / 2 + margin
        top = config.frame_height / 2 - margin
        x0, x1, y0, y1 = self._aabb_xy(m, pad=0.0)
        return (x0 >= left and x1 <= right and y0 >= bottom and y1 <= top)

    def bbox_intersects(self, a: Mobject, b: Mobject, pad: float = 0.14) -> bool:
        ax0, ax1, ay0, ay1 = self._aabb_xy(a, pad=pad)
        bx0, bx1, by0, by1 = self._aabb_xy(b, pad=pad)
        return not (ax1 < bx0 or bx1 < ax0 or ay1 < by0 or by1 < ay0)

    def fit_to_safe_width(self, m: Mobject) -> Mobject:
        safe_w = config.frame_width - 1.0
        if m.width > safe_w:
            m.scale(safe_w / m.width)
        return m

    def build_subtitle(self, text: str) -> VGroup:
        s = self.normalize_one_line(text)
        t = Text(s, color=WHITE).scale(0.55)
        bg = Rectangle(height=t.height + 0.25, width=min(config.frame_width - 1.0, t.width + 0.6))
        bg.set_fill(BLACK, opacity=0.55).set_stroke(width=0)
        g = VGroup(bg, t)
        t.move_to(bg.get_center())
        g.to_edge(DOWN, buff=0.5)
        if g.width > (config.frame_width - 1.0):
            g.scale((config.frame_width - 1.0) / g.width)
        return g

    def make_text(self, kind: str, text: str, scale_hint: float = 1.0) -> Mobject:
        s = self.normalize_one_line(text)
        base = {"title": 0.85, "corner": 0.55, "caption": 0.55}.get(kind, 0.6)
        m = Text(s, color=WHITE).scale(base * scale_hint)  # stable demo
        return self.fit_to_safe_width(m)

    def place_text(self, m: Mobject, anchor: Optional[Mobject], avoid: List[Mobject], prefer: str,
                   min_gap: float = 0.3, margin: float = 0.5) -> Mobject:
        pad = 0.14

        def accept() -> bool:
            if not self.inside_safe_area(m, margin):
                return False
            return all(not self.bbox_intersects(m, x, pad) for x in avoid if x is not m)

        # 1) shrink first
        self.fit_to_safe_width(m)
        for _ in range(5):
            if accept():
                return m
            m.scale(0.92)

        # 2) base candidates
        if prefer == "title":
            m.to_edge(UP, buff=0.5)
            if accept():
                return m
        if prefer == "bottom":
            m.to_edge(DOWN, buff=0.5)
            if accept():
                return m

        if anchor is not None:
            for d in [DOWN, UP, LEFT, RIGHT]:
                m.next_to(anchor, d, buff=0.35)
                if accept():
                    return m

        for corner in [UR, UL, DR, DL]:
            m.to_corner(corner, buff=0.5)
            if accept():
                return m

            # 3) diagonal deep search (two steps)
            for step, max_steps in [(0.32, 8), (0.22, 10)]:
                for dv in [(UP + RIGHT), (UP + LEFT), (DOWN + RIGHT), (DOWN + LEFT)]:
                    for _ in range(max_steps):
                        m.shift(dv * step)
                        if accept():
                            return m

            # 4) 8-dir deep search (two steps)
            for step, max_steps in [(0.32, 8), (0.22, 10)]:
                for dv in [UP + RIGHT, UP + LEFT, DOWN + RIGHT, DOWN + LEFT, UP, DOWN, LEFT, RIGHT]:
                    for _ in range(max_steps):
                        m.shift(dv * step)
                        if accept():
                            return m

        # 5) final fallback
        m.scale(0.95)
        # TODO(CONSTRAINT): auto-placement failed; overlap risk remains
        return m


class RepairPromptDemo(VoiceoverScene):
    def construct(self):

        helper = Scene_Helper(self)
        self._helper = helper

        self._persist: List[Mobject] = []

    def slot_put(self, slot: str, new: Mobject, method: str = "transform") -> Mobject:
        old = self._slots.get(slot)
        new = self._helper.place_text(
            new,
            anchor=None,
            avoid=list(self._slots.values()) + self._persist,
            prefer=("title" if slot == "title" else "bottom" if slot == "subtitle" else "corner_ur"),
        )
        if old is None:
            self.play(FadeIn(new), run_time=0.35)
            self._slots[slot] = new
            self._auto_scan(self._helper, "after_slot", True)
            return new
        if method == "transform":
            self.play(Transform(old, new), run_time=0.35)
            self._slots[slot] = old
            self._auto_scan(self._helper, "after_slot", True)
            return old
        self.play(FadeOut(old), FadeIn(new), run_time=0.35)
        self._slots[slot] = new
        self._auto_scan(self._helper, "after_slot", True)
        return new

    def slot_clear(self, slot: str, fade: bool = True) -> None:
        old = self._slots.get(slot)
        if old is None:
            return
        self.next_section()
        if fade:
            self.play(FadeOut(old), run_time=0.3)
        else:
            self.remove(old)
        del self._slots[slot]

    def temp_show(self, m: Mobject, anim: str = "fade", rt: float = 0.6) -> None:
        if m not in self._temps:
            self._temps.append(m)
        m = self._helper.place_text(m, anchor=None, avoid=list(self._slots.values()) + self._persist, prefer="corner_dr")
        if anim == "write":
            self.play(Write(m), run_time=rt)
        else:
            self.play(FadeIn(m), run_time=rt)
        self._auto_scan(self._helper, "after_temp", True)

    def temp_clear(self, fade: bool = True) -> None:
        if not self._temps:
            return
        self.next_section()
        if fade:
            self.play(*[FadeOut(x) for x in self._temps], run_time=0.45)
        else:
            self.remove(*self._temps)
        self._temps = []

    def _auto_scan(self, helper: Scene_Helper, scope: str, try_fix: bool = True) -> None:
        all_objs = list(self._slots.values()) + list(self._temps) + list(self._persist)
        text_like = [m for m in (self._stage_local_texts + list(self._slots.values())) if isinstance(m, (Text, Tex, MathTex, VGroup))]
        bad: List[Mobject] = []
        for t in text_like:
            if not helper.inside_safe_area(t, 0.5):
                bad.append(t)
                continue
            for other in all_objs:
                if other is t:
                    continue
                if helper.bbox_intersects(t, other, 0.14):
                    bad.append(t)
                    break
        if bad and try_fix:
            self._rebuild_fix(helper, bad, f"auto-scan:{scope}")

    def _rebuild_fix(self, helper: Scene_Helper, targets: List[Mobject], reason: str) -> None:
        avoid = list(self._slots.values()) + list(self._persist)

        # Step1: re-place + shrink (max 2)
        for t in targets:
            for _ in range(2):
                helper.place_text(t, None, avoid, "bottom")
                if helper.inside_safe_area(t, 0.5) and all(not helper.bbox_intersects(t, x, 0.14) for x in avoid if x is not t):
                    return
                t.scale(0.95)

        # Step2: zone fallback
        for t in targets:
            for pref in ["bottom", "corner_dr", "corner_ur", "corner_dl", "corner_ul"]:
                helper.place_text(t, None, avoid, pref)
                if helper.inside_safe_area(t, 0.5) and all(not helper.bbox_intersects(t, x, 0.14) for x in avoid if x is not t):
                    return

        # Step3: slot rebuild
        slot_by_id = {id(v): k for k, v in self._slots.items()}
        for t in targets:
            slot = slot_by_id.get(id(t))
            if slot is None:
                continue
            raw = getattr(t, "text", None) or getattr(t, "tex_string", None) or str(t)
            rebuilt = Text(helper.normalize_one_line(raw), color=WHITE).scale(0.6)
            rebuilt = helper.place_text(rebuilt, None, avoid, "bottom" if slot == "subtitle" else "title" if slot == "title" else "corner_ur")
            self.slot_put(slot, rebuilt, method="fade")
            # TODO(CONSTRAINT): forced rebuild to prevent stacking
            return

        # Step4: temp rebuild
        if self._temps:
            temps_snapshot = list(self._temps)
            self.temp_clear(fade=True)
            for t in temps_snapshot:
                if t not in targets:
                    continue
                rebuilt = t.copy()
                rebuilt = helper.place_text(rebuilt, None, list(self._slots.values()) + self._persist, "corner_dr")
                self.temp_show(rebuilt, anim="fade", rt=0.25)
            return

        # Step5: Tex/MathTex -> Text
        for t in targets:
            if isinstance(t, (Tex, MathTex)):
                raw = getattr(t, "tex_string", None) or str(t)
                downgraded = Text(helper.normalize_one_line(raw), color=WHITE).scale(0.6)
                downgraded = helper.place_text(downgraded, None, avoid, "bottom")
                self.temp_show(downgraded, anim="fade", rt=0.25)
                # TODO(CONSTRAINT): forced Text fallback for layout stability
                return

        # TODO(CONSTRAINT): overlap persists even after REBUILD FIX; manual layout needed

    def _swap_title(self, helper: Scene_Helper, text: str) -> None:
        t = helper.make_text("title", text)
        t = helper.place_text(t, None, list(self._slots.values()) + self._persist, "title")
        self._stage_local_texts.append(t)
        self._title_mobj = self.slot_put("title", t, method="transform")

    def stage_scene1(self, helper: Scene_Helper):
        self._stage_local_texts = []
        self.temp_clear()
        self._auto_scan(helper, "stage_begin", True)

        self.next_section()

        anchor = Square().set_stroke(BLUE, 4).set_fill(opacity=0).scale(1.0)

        self.play(Create(anchor), run_time=0.8)
        self._persist.append(anchor)

        MIN_T = 1.2
        raw_vo = "we show one message and remove it before the next stage"
        vo_text = helper.fix_transcript(raw_vo)
        with self.voiceover(text=vo_text) as tracker:
            sub = helper.build_subtitle(vo_text)
            sub = helper.place_text(sub, None, list(self._slots.values()) + self._persist, "bottom")
            self._stage_local_texts.append(sub)
            self.slot_put("subtitle", sub, method="fade")

            rt = max(MIN_T, float(getattr(tracker, "duration", MIN_T)))
            msg = Text("Temporary message", color=YELLOW).scale(0.55)

            self._stage_local_texts.append(msg)

        self._auto_scan(helper, "stage_end", True)
