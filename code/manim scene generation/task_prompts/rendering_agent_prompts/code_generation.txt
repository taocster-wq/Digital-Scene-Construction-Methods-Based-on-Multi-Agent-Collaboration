````
## 2) 代码例子骨架（示范结构；真实生成时要替换为你的 scene_plan/scene_implementation 内容）

> 这是“例子骨架”，你可以放进提示词里作为示范，也可以单独给模型参考。
> 重点：**完全不使用 get_bounding_box / bounding_box**，全部 bbox 走 `get_all_points()` AABB（回退 center+width/height）。

```python
from __future__ import annotations

from manim import *
from typing import Optional, List, Dict, Callable, Tuple
import re
import numpy as np

import sys, os
sys.path.append(os.path.dirname(os.path.dirname(os.path.abspath(__file__))))
from utils.kokoro_voiceover import KokoroService

try:
    from manim_voiceover import VoiceoverScene
    HAS_VO = True
except Exception:
    VoiceoverScene = Scene
    HAS_VO = False

USE_LATEX = True
DEBUG_LAYOUT = False


class Scene_Helper:
    def __init__(self, scene: Scene):
        self.scene = scene

    # ---------- transcript fix (YouTube-like, do NOT change word choice) ----------
    def fix_transcript(self, text: str) -> str:
        s = (text or "").replace("\n", " ").strip()
        s = " ".join(s.split())
        if not s:
            return s
        if s[0].isalpha():
            s = s[0].upper() + s[1:]
        if s[-1].isalnum():
            s += "."
        s = re.sub(r"([,.!?])(\w)", r"\1 \2", s)
        return s

    def normalize_one_line(self, text: str) -> str:
        return " ".join((text or "").replace("\n", " ").split())

    # ---------- robust AABB (ManimCE v0.19.0 safe) ----------
    def _aabb_xy(self, m: Mobject, pad: float = 0.0) -> Tuple[float, float, float, float]:
        # NEVER use get_bounding_box()/bounding_box in ManimCE v0.19.0
        try:
            pts = np.array(m.get_all_points())
        except Exception:
            pts = np.empty((0, 3))

        if pts.size == 0:
            c = np.array(m.get_center())
            w = float(getattr(m, "width", 0.0))
            h = float(getattr(m, "height", 0.0))
            x0, x1 = float(c[0] - w / 2), float(c[0] + w / 2)
            y0, y1 = float(c[1] - h / 2), float(c[1] + h / 2)
        else:
            x0, x1 = float(pts[:, 0].min()), float(pts[:, 0].max())
            y0, y1 = float(pts[:, 1].min()), float(pts[:, 1].max())

        return (x0 - pad, x1 + pad, y0 - pad, y1 + pad)

    def safe_rect(self) -> Rectangle:
        r = Rectangle(width=config.frame_width - 1.0, height=config.frame_height - 1.0)
        r.set_stroke(GREY_B, width=1).set_fill(opacity=0)
        return r

    def inside_safe_area(self, m: Mobject, margin: float = 0.5) -> bool:
        left = -config.frame_width / 2 + margin
        right = config.frame_width / 2 - margin
        bottom = -config.frame_height / 2 + margin
        top = config.frame_height / 2 - margin
        x0, x1, y0, y1 = self._aabb_xy(m, pad=0.0)
        return (x0 >= left and x1 <= right and y0 >= bottom and y1 <= top)

    def bbox_intersects(self, a: Mobject, b: Mobject, pad: float = 0.14) -> bool:
        ax0, ax1, ay0, ay1 = self._aabb_xy(a, pad=pad)
        bx0, bx1, by0, by1 = self._aabb_xy(b, pad=pad)
        return not (ax1 < bx0 or bx1 < ax0 or ay1 < by0 or by1 < ay0)

    def fit_to_safe_width(self, m: Mobject) -> Mobject:
        safe_w = config.frame_width - 1.0
        if m.width > safe_w:
            m.scale(safe_w / m.width)
        return m

    # ---------- text factory ----------
    def _tex_or_text(self, s: str) -> Mobject:
        if not USE_LATEX:
            return Text(s)
        # simple heuristic: math-ish -> MathTex, else Tex
        if re.search(r"[=+\-*/^\\]", s):
            return MathTex(s)
        return Tex(s)

    def make_text(self, kind: str, text: str, scale_hint: float = 1.0) -> Mobject:
        s = self.normalize_one_line(text)
        base = {
            "title": 0.85,
            "corner": 0.55,
            "step": 0.65,
            "caption": 0.55,
            "label": 0.60,
        }.get(kind, 0.6)
        m = self._tex_or_text(s)
        if isinstance(m, (Text, Tex, MathTex)):
            m.set_color(WHITE)
        m.scale(base * scale_hint)
        return self.fit_to_safe_width(m)

    def build_subtitle(self, text: str) -> VGroup:
        s = self.normalize_one_line(text)
        t = Text(s, color=WHITE).scale(0.55)
        bg = Rectangle(height=t.height + 0.25, width=min(config.frame_width - 1.0, t.width + 0.6))
        bg.set_fill(BLACK, opacity=0.55).set_stroke(width=0)
        g = VGroup(bg, t)
        t.move_to(bg.get_center())
        g.to_edge(DOWN, buff=0.5)
        if g.width > (config.frame_width - 1.0):
            g.scale((config.frame_width - 1.0) / g.width)
        return g

    # ---------- placement / anti-overlap ----------
    def _accept(self, m: Mobject, avoid: List[Mobject], margin: float, pad: float) -> bool:
        if not self.inside_safe_area(m, margin=margin):
            return False
        for x in avoid:
            if x is m:
                continue
            if self.bbox_intersects(m, x, pad=pad):
                return False
        return True

    def _candidate_fns(self, m: Mobject, anchor: Optional[Mobject], prefer: str) -> List[Callable[[], None]]:
        c: List[Callable[[], None]] = []
        if prefer == "title":
            c.append(lambda: m.to_edge(UP, buff=0.5))
        if prefer == "bottom":
            c.append(lambda: m.to_edge(DOWN, buff=0.5))
        if prefer == "corner_ur":
            c.append(lambda: m.to_corner(UR, buff=0.5))
        if prefer == "corner_ul":
            c.append(lambda: m.to_corner(UL, buff=0.5))
        if prefer == "corner_dr":
            c.append(lambda: m.to_corner(DR, buff=0.5))
        if prefer == "corner_dl":
            c.append(lambda: m.to_corner(DL, buff=0.5))

        if anchor is not None:
            c.extend([
                lambda: m.next_to(anchor, DOWN, buff=0.35),
                lambda: m.next_to(anchor, UP, buff=0.35),
                lambda: m.next_to(anchor, LEFT, buff=0.35),
                lambda: m.next_to(anchor, RIGHT, buff=0.35),
            ])

        c.extend([
            lambda: m.to_corner(UR, buff=0.5),
            lambda: m.to_corner(UL, buff=0.5),
            lambda: m.to_corner(DR, buff=0.5),
            lambda: m.to_corner(DL, buff=0.5),
        ])
        return c

    def _deep_search(self, m: Mobject, avoid: List[Mobject], margin: float, pad: float,
                     dirs: List[np.ndarray], step: float, max_steps: int) -> bool:
        for dvec in dirs:
            for _ in range(max_steps):
                m.shift(dvec * step)
                if self._accept(m, avoid, margin, pad):
                    return True
        return False

    def _least_overlap_score(self, m: Mobject, avoid: List[Mobject], pad: float) -> int:
        cnt = 0
        for x in avoid:
            if x is m:
                continue
            if self.bbox_intersects(m, x, pad=pad):
                cnt += 1
        return cnt

    def place_text(self, m: Mobject, anchor: Optional[Mobject], avoid: List[Mobject], prefer: str,
                   min_gap: float = 0.3, margin: float = 0.5) -> Mobject:
        pad = 0.14
        self.fit_to_safe_width(m)

        # (A) shrink up to 5
        for _ in range(5):
            if self._accept(m, avoid, margin, pad):
                return m
            m.scale(0.92)

        candidates = self._candidate_fns(m, anchor, prefer)
        diag_dirs = [UP + RIGHT, UP + LEFT, DOWN + RIGHT, DOWN + LEFT]
        eight_dirs = [UP + RIGHT, UP + LEFT, DOWN + RIGHT, DOWN + LEFT, UP, DOWN, LEFT, RIGHT]

        # (B)(C)(D)
        for fn in candidates:
            fn()
            if self._accept(m, avoid, margin, pad):
                return m
            if self._deep_search(m, avoid, margin, pad, diag_dirs, 0.32, 8):
                return m
            if self._deep_search(m, avoid, margin, pad, diag_dirs, 0.22, 10):
                return m
            if self._deep_search(m, avoid, margin, pad, eight_dirs, 0.32, 8):
                return m
            if self._deep_search(m, avoid, margin, pad, eight_dirs, 0.22, 10):
                return m

        # (F) final fallback: least overlap among corners
        best_corner = UR
        best_score = 10**9
        for corner in [UR, DR, UL, DL]:
            test = m.copy()
            test.to_corner(corner, buff=0.5)
            score = self._least_overlap_score(test, avoid, pad)
            if score < best_score:
                best_score = score
                best_corner = corner
        m.to_corner(best_corner, buff=0.5)
        # TODO(CONSTRAINT): auto-placement failed; overlap risk remains
        return m

    def debug_overlay(self, named: List[Tuple[str, Mobject]]) -> VGroup:
        g = VGroup()
        if not DEBUG_LAYOUT:
            return g
        g.add(self.safe_rect())
        for name, m in named:
            try:
                box = SurroundingRectangle(m, buff=0.02)
                box.set_stroke(YELLOW, width=1)
                tag = Text(name, color=YELLOW).scale(0.35)
                tag.next_to(box, UP, buff=0.05)
                g.add(box, tag)
            except Exception:
                pass
        return g


class GeneratedScene(VoiceoverScene):
    # NOTE: in real generation, class name must be auto-derived from topic/description.
    def construct(self):
        helper = Scene_Helper(self)
        self._helper = helper

        self._persist: List[Mobject] = []
        self._temps: List[Mobject] = []
        self._slots: Dict[str, Mobject] = {}
        self._title_mobj: Optional[Mobject] = None
        self._stage_local_texts: List[Mobject] = []

        # voiceover service (only when narration non-empty in real generation)

            self.set_speech_service(KokoroService())
        except Exception:
            pass

        # Example stage order
        self.stage_scene1(helper)
        self.stage_scene2(helper)

    # ---------------- SLOT / TEMP ----------------
    def slot_put(self, slot: str, m: Mobject, method: str = "transform") -> Mobject:
        old = self._slots.get(slot)
        prefer = "title" if slot == "title" else "bottom" if slot == "subtitle" else "corner_ur"
        m = self._helper.place_text(m, anchor=None, avoid=list(self._slots.values()) + self._persist, prefer=prefer)

        if old is None:
            self.play(FadeIn(m), run_time=0.35)
            self._slots[slot] = m
            self._auto_scan(self._helper, scope="after_slot", try_fix=True)
            return m

        if method == "transform":
            self.play(Transform(old, m), run_time=0.35)
            self._slots[slot] = old
            self._auto_scan(self._helper, scope="after_slot", try_fix=True)
            return old

        self.play(FadeOut(old), FadeIn(m), run_time=0.35)
        self._slots[slot] = m
        self._auto_scan(self._helper, scope="after_slot", try_fix=True)
        return m

    def slot_clear(self, slot: str, fade: bool = True) -> None:
        old = self._slots.get(slot)
        if old is None:
            return
        self.next_section()
        if fade:
            self.play(FadeOut(old), run_time=0.3)
        else:
            self.remove(old)
        del self._slots[slot]

    def temp_show(self, m: Mobject, anim: str = "fade", rt: float = 0.6) -> None:
        if m not in self._temps:
            self._temps.append(m)
        m = self._helper.place_text(m, anchor=None, avoid=list(self._slots.values()) + self._persist, prefer="corner_dr")
        if anim == "write":
            self.play(Write(m), run_time=rt)
        else:
            self.play(FadeIn(m), run_time=rt)
        self._auto_scan(self._helper, scope="after_temp", try_fix=True)

    def temp_clear(self, fade: bool = True) -> None:
        if not self._temps:
            return
        self.next_section()
        if fade:
            self.play(*[FadeOut(x) for x in self._temps], run_time=0.45)
        else:
            self.remove(*self._temps)
        self._temps = []

    # ---------------- AUTO SCAN ----------------
    def _auto_scan(self, helper: Scene_Helper, scope: str, try_fix: bool = True) -> None:
        all_objs = list(self._slots.values()) + list(self._temps) + list(self._persist)
        text_like: List[Mobject] = []
        for m in list(self._stage_local_texts) + list(self._slots.values()) + list(self._temps):
            if isinstance(m, (Text, Tex, MathTex, VGroup)):
                text_like.append(m)

        bad: List[Mobject] = []
        for t in text_like:
            if not helper.inside_safe_area(t, margin=0.5):
                bad.append(t)
                continue
            for other in all_objs:
                if other is t:
                    continue
                if helper.bbox_intersects(t, other, pad=0.14):
                    bad.append(t)
                    break

        if bad and try_fix:
            self._rebuild_fix(helper, bad, reason=f"auto-scan:{scope}")
            # re-scan once (silent)

    # ---------------- REBUILD FIX (Step1~Step5) ----------------
    def _rebuild_fix(self, helper: Scene_Helper, targets: List[Mobject], reason: str) -> None:
        avoid_base = list(self._slots.values()) + list(self._persist)

        slot_of: Dict[int, str] = {id(v): k for k, v in self._slots.items()}
        temps_snapshot = list(self._temps)

        def ok(m: Mobject) -> bool:
            if not helper.inside_safe_area(m, margin=0.5):
                return False
            for x in avoid_base:
                if x is m:
                    continue
                if helper.bbox_intersects(m, x, pad=0.14):
                    return False
            return True

        # Step1: re-place + shrink (max 2)
        for t in targets:
            for _ in range(2):
                helper.place_text(t, anchor=None, avoid=avoid_base, prefer="bottom")
                if ok(t):
                    return
                t.scale(0.95)

        # Step2: force prefer zones
        prefs = ["bottom", "corner_dr", "corner_ur", "corner_dl", "corner_ul"]
        for t in targets:
            for p in prefs:
                helper.place_text(t, anchor=None, avoid=avoid_base, prefer=p)
                if ok(t):
                    return

        # Step3: SLOT rebuild
        for t in targets:
            s = slot_of.get(id(t))
            if s is None:
                continue
            raw = getattr(t, "text", None)
            if raw is None:
                raw = getattr(t, "tex_string", None)
            if raw is None:
                raw = str(t)
            rebuilt = helper.make_text("caption" if s != "title" else "title", helper.normalize_one_line(raw), 1.0)
            rebuilt = helper.place_text(rebuilt, anchor=None, avoid=[x for x in avoid_base if x is not t],
                                        prefer=("title" if s == "title" else "bottom" if s == "subtitle" else "corner_ur"))
            self.slot_put(s, rebuilt, method="fade")
            return

        # Step4: TEMP clear and rebuild minimal
        if temps_snapshot:
            try:
                self.temp_clear(fade=True)
            except Exception:
                self._temps = []
            for t in temps_snapshot:
                if t not in targets:
                    continue
                rebuilt = t.copy()
                rebuilt = helper.place_text(rebuilt, anchor=None, avoid=list(self._slots.values()) + self._persist, prefer="corner_dr")
                self.temp_show(rebuilt, anim="fade", rt=0.25)
            return

        # Step5: type downgrade Tex/MathTex -> Text
        for t in targets:
            if isinstance(t, (Tex, MathTex)):
                raw = getattr(t, "tex_string", None)
                if raw is None:
                    raw = str(t)
                downgraded = Text(helper.normalize_one_line(raw), color=WHITE).scale(0.58)
                downgraded = helper.place_text(downgraded, anchor=None, avoid=avoid_base, prefer="bottom")
                s = slot_of.get(id(t))
                if s is not None:
                    self.slot_put(s, downgraded, method="fade")
                else:
                    self.temp_show(downgraded, anim="fade", rt=0.25)
                # TODO(CONSTRAINT): layout forced fallback Text for stability
                return

        # TODO(CONSTRAINT): overlap persists even after REBUILD FIX; manual layout needed
````
